
\begin{problem}
Determine (using the inclusion-exclusion principle)
the number of integer solutions of the equation:
%
\begin{eqnarray*}
x + y + z &=& 20,
\end{eqnarray*}
%
under the constraints 
%
\begin{eqnarray*}
        1 \;\le\; x \;\le\; 5 \\
        3 \;\le\; y \;\le\; 9 \\
        1\;\le\; z  \;\le\; 8
\end{eqnarray*}
%
Show your work.
\end{problem}
